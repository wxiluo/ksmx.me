#include
main()
{
	int x = 8;
	int *q, *p = &x;

	q = p;

	printf("整形变量x的值是:%d\n", x);
	printf("指针p所指向的目标值*p是:%d\n", *p);
	printf("指针q所指向的目标值*p是:%d\n\n", *q);
	printf("变量x的地址&x是:%xH\n", &x);
	printf("指针p中的值是:%xH\n", p);
	printf("指针q中的值是:%xH\n", q);
	printf("指针p的地址&p是:%xH\n", &p);
	printf("指针q的地址&q是:%xH\n", &q);
}
